167  struct PDFKernelWrapper
168  {
169  const Distribution left_;
170  const Distribution right_;
171  const Scalar x_;
172  const Bool isZero_;
173  const Scalar pdf0_;
174 
175  PDFKernelWrapper(const Distribution & left,
176  const Distribution & right,
177  const Scalar x):
178  left_(left), right_(right), x_(x), isZero_(std::abs(x) < ResourceMap::GetAsScalar("Distribution-DefaultQuantileEpsilon")), pdf0_(isZero_ ? right.computePDF(0.0) : 0.0) {};
179 
180  Point eval(const Point & point) const
181  {
182  const Scalar value = left_.computePDF(point);
183  if (value == 0.0) return Point(1, 0.0);
184  const Scalar u = point[0];
185  const Scalar absU = std::abs(u);
186  // x_ == 0
187  if (isZero_)
188  {
189  if (pdf0_ == 0.0) return Point(1, 0.0);
190  if (absU == 0.0) return Point(1, SpecFunc::MaxScalar);
191  return Point(1, value * pdf0_ / absU);
192  }
193  // x_ != 0
194  if (absU == 0.0)
195  {
196  const Scalar epsilon = 1e-7;
197  return Point(1, value * 0.5 * (right_.computePDF(x_ / epsilon) + right_.computePDF(-x_ / epsilon)) / epsilon);
198  }
199  return Point(1, value * right_.computePDF(x_ / u) / absU);
200  };
