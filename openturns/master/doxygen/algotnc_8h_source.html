94 /*
95  * A function as required by tnc
96  * state is a void pointer provided to the function at each call
97  *
98  * x : on input, then vector of variables (should not be modified)
99  * f : on output, the value of the function
100  * g : on output, the value of the gradient
101  * state : on input, the value of the state variable as provided to tnc
102  *
103  * must returns 0 if no error occurs or 1 to immediately end the minimization.
104  *
105  */
106 typedef int tnc_function(double x[], double *f, double g[], void *state);
107 
108 /*
109  * tnc : minimize a function with variables subject to bounds, using
110  * gradient information.
111  *
112  * n : number of variables (must be >= 0)
113  * x : on input, initial estimate ; on output, the solution
114  * f : on output, the function value at the solution
115  * g : on output, the gradient value at the solution
116  * g should be an allocated vector of size n or NULL,
117  * in which case the gradient value is not returned.
118  * function : the function to minimize (see tnc_function)
119  * state : used by function (see tnc_function)
120  * low, up : the bounds
121  * set low[i] to -HUGE_VAL to remove the lower bound
122  * set up[i] to HUGE_VAL to remove the upper bound
123  * if low == NULL, the lower bounds are removed.
124  * if up == NULL, the upper bounds are removed.
125  * scale : scaling factors to apply to each variable
126  * if NULL, the factors are up-low for interval bounded variables
127  * and 1+|x] for the others.
128  * offset : constant to substract to each variable
129  * if NULL, the constant are (up+low)/2 for interval bounded
130  * variables and x for the others.
131  * messages : see the tnc_message enum
132  * maxCGit : max. number of hessian*vector evaluation per main iteration
133  * if maxCGit == 0, the direction chosen is -gradient
134  * if maxCGit < 0, maxCGit is set to max(1,min(50,n/2))
135  * maxnfeval : max. number of function evaluation
136  * eta : severity of the line search. if < 0 or > 1, set to 0.25
137  * stepmx : maximum step for the line search. may be increased during call
138  * if too small, will be set to 10.0
139  * accuracy : relative precision for finite difference calculations
140  * if <= machine_precision, set to sqrt(machine_precision)
141  * fmin : minimum function value estimate
142  * ftol : precision goal for the value of f in the stoping criterion
143  * if ftol < 0.0, ftol is set to accuracy
144  * xtol : precision goal for the value of x in the stopping criterion
145  * (after applying x scaling factors)
146  * if xtol < 0.0, xtol is set to sqrt(machine_precision)
147  * pgtol : precision goal for the value of the projected gradient in the
148  * stopping criterion (after applying x scaling factors)
149  * if pgtol < 0.0, pgtol is set to 1e-2 * sqrt(accuracy)
150  * setting it to 0.0 is not recommended
151  * rescale : f scaling factor (in log10) used to trigger f value rescaling
152  * if 0, rescale at each iteration
153  * if a big value, never rescale
154  * if < 0, rescale is set to 1.3
155  * nfeval : on output, the number of function evaluations.
156  * ignored if nfeval==NULL.
157  *
158  * The tnc function returns a code defined in the tnc_rc enum.
159  * On output, x, f and g may be very slightly out of sync because of scaling.
160  *
161  */
162 extern int tnc(int n, double x[], double *f, double g[],
163  tnc_function *function, void *state,
164  double low[], double up[], double scale[], double offset[],
165  int messages, int maxCGit, int maxnfeval, double eta, double stepmx,
166  double accuracy, double fmin, double ftol, double xtol, double pgtol,
167  double rescale, int *nfeval);
