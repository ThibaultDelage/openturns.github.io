201  adv.saveAttribute( "size", this->getSize() );
202  // We cannot write
203  // std::copy( Collection<T>::coll__.begin(),
204  // Collection<T>::coll__.end(),

208  for (const_iterator cit = Collection<T>::coll__.begin();
209  cit != Collection<T>::coll__.end();
