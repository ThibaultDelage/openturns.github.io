41 // In order to create an H-matrix, one has first to generate an interface with
42 // HMatrixFactory::build, then compute its coefficients, and after that, it is
43 // possible to factorize it, solve linear systems, etc. As H-matrices are
44 // designed to solve very large systems, most operations are performed inplace,
45 // which is an unusual pattern in OpenTURNS.
46 
47 // There are two ways to compute matrix coefficients:
48 // 1. The simplest solution is to provide a method which computes (i,j) coefficient.
49 // This is what HMatrixRealAssemblyFunction is designed for, and such an object
50 // is then passed to HMatrixImplementation::simpleAssemble. But most of the time,
51 // this involves heavy computations which could be reused when computing values
52 // for different coefficients.
53 // 2. HMat provides a 2nd interface to compute tiled blocks. But it is cumbersome,
54 // so we defined a more intuitive interface which should be suited for our needs.
55 // When dimension is greater than 1, HMatrixFactory::build duplicates input vertices
56 // to have local d x d matrices, and HMatrixTensorRealAssemblyFunction is an interface
57 // to compute all coefficients of this local matrix, which had already been allocated.
58 // It will be automatically copied at the right place in HMatrixImplementation.

197 // First implementation, by using HMatrixRealAssemblyFunction
198 // This is going to be very slow, because each local
199 // covariance matrix of size 3x3 is computed, but a single
200 // coefficient is stored.
201 class CovarianceAssemblyFunction : public HMatrixRealAssemblyFunction
202 {
203 private:
204  const CovarianceModel covarianceModel_;
205  const Sample vertices_;
206  const UnsignedInteger covarianceDimension_;
207  const double epsilon_;
208 
209 public:
210  CovarianceAssemblyFunction(const CovarianceModel & covarianceModel, const Sample & vertices, double epsilon)
211  : HMatrixRealAssemblyFunction()
212  , covarianceModel_(covarianceModel)
213  , vertices_(vertices)
214  , covarianceDimension_(covarianceModel.getOutputDimension())
215  , epsilon_(epsilon) {}
216 
217  Scalar operator()(UnsignedInteger i, UnsignedInteger j) const
218  {
219  const UnsignedInteger rowIndex = i / covarianceDimension_;
220  const UnsignedInteger columnIndex = j / covarianceDimension_;
221  const CovarianceMatrix localCovarianceMatrix(covarianceModel_( vertices_[rowIndex], vertices_[columnIndex] ));
222  const UnsignedInteger rowIndexLocal = i % covarianceDimension_;
223  const UnsignedInteger columnIndexLocal = j % covarianceDimension_;
224  return localCovarianceMatrix(rowIndexLocal, columnIndexLocal) + (i != j ? 0.0 : epsilon_);
225  }
226 };

228 // Second implementation, by using HMatrixTensorRealAssemblyFunction
229 // Each local covariance matrix is built only once, and its components
230 // are dispatched into the global covariance matrix
231 class CovarianceBlockAssemblyFunction : public HMatrixTensorRealAssemblyFunction
232 {
233 private:
234  const CovarianceModel covarianceModel_;
235  const Sample vertices_;
236  const double epsilon_;
237  CovarianceMatrix epsilonId_;
238 
239 public:
240  CovarianceBlockAssemblyFunction(const CovarianceModel & covarianceModel, const Sample & vertices, double epsilon)
241  : HMatrixTensorRealAssemblyFunction(covarianceModel.getOutputDimension())
242  , covarianceModel_(covarianceModel)
243  , vertices_(vertices)
244  , epsilon_(epsilon)
245  {
246  Matrix eps = epsilon_ * IdentityMatrix(covarianceModel.getOutputDimension());
247  Pointer<MatrixImplementation> impl = eps.getImplementation();
248  epsilonId_ = CovarianceMatrix(covarianceModel.getOutputDimension(), *impl.get());
249  }
250 
251  void compute(UnsignedInteger i, UnsignedInteger j, Matrix* localValues) const
252  {
253  CovarianceMatrix localResult(covarianceModel_( vertices_[i], vertices_[j] ));
254  if (i == j && epsilon_ != 0.0)
255  localResult = localResult + epsilonId_;
256  memcpy( &localValues->getImplementation()->operator[](0), &localResult.getImplementation()->operator[](0), dimension_ * dimension_ * sizeof(Scalar) );
257  }
258 };
