139  // Compute |u|p_left(u)p_right(ux) where x is the argument of the pdf of left*right
140  Point eval(const Point & point) const
141  {
142  const Scalar u = point[0];
143  const Scalar absU = std::abs(u);
144  // First special case: |u|==0
145  if (absU == 0.0) return Point(1, 0.0);
146  // Secon special case: x==0
147  if (isZero_)
148  {
149  if (pdf0_ == 0.0) return Point(1, 0.0);
150  return Point(1, absU * pdf0_ * left_.computePDF(point));
151  }
152  const Scalar value = left_.computePDF(point);
153  // If the given point is outside of the support of left
154  if (value == 0.0) return Point(1, 0.0);
155  return Point(1, absU * right_.computePDF(u * x_) * value);
156  };
